Here are my advances related to research during my PhD:

1. Statistical software

[Retrieving data]

Using the packages quandl and BIS, I’ve learned techniques to retrieve data online. The first is used to collect financial markets data, while the second offers a direct connection to the data supplied by the Bank for International Settlements.

Example of code:

library(quantmod)
library(dplyr)

# DATA FROM YAHOO FINANCE
# IBOVESPA (^BVSP)

getSymbols("^BVSP",src="yahoo", from="1900-01-01")
BVSP.DF <- data.frame(Date=index(BVSP),coredata(BVSP))
BVSP.DF <- dplyr::select(BVSP.DF,Date, BVSP.Close)
plot(BVSP.DF)


library(devtools)
library(BIS)
library(dplyr)
library(ggplot2)

# DATA FROM BIS
# POLICY RATES (INTEREST RATES)

datasets <- get_datasets()
head(datasets, 23)
prates <- get_bis(datasets$url[23], quiet = TRUE)
BRLIR <- subset(prates, ref_area == "BR")
BRLIR <- dplyr::select(BRLIR, date, obs_value)
names(BRLIR)[names(BRLIR) == "obs_value"] <- "BRLIR"
BRLIR <- BRLIR[BRLIR$date >= "2000-01-01" & BRLIR$date <= "2020-02-03",]
BIR <- ggplot(BRLIR, aes(x=date, y=BRLIR)) +
  geom_line() + 
  xlab("")
BIR

[Programming]

I’ve learned to code .do files, like this:

clear all
log using "resultats_`c(current_date)'.log", replace
capture log
set more off
global data "C:\Data\"
cd "$data\"
import excel "C:\Data\CHF.data.xlsx", firstrow
format Obs %tw
tsset Obs

*** Generate new variables

g CT = Short/Long
mvencode CT, mv(0)
g IRD = IR - USIR

*** Generate variables in first differences

gen dER = d.ER

*** Generate variables in log

g lER = log(ER)

*** Data manipulation

drop if Obs==1

*** Group of variables

global levelvar ER CT IRD VIX SM SMUS
global logvar lER lCT IRD lVIX lSM lSMUS

*** Graphics

foreach var of varlist $levelvar {
quietly: line `var' Obs, saving(level`var', replace)
quietly: graph export "`var'-level.pdf", replace
}
quietly: gr combine levelER.gph levelCT.gph levelIRD.gph levelVIX.gph levelSM.gph levelSMUS.gph, saving(level_all, replace)
quietly: graph export "level_all.pdf", replace

*** Generate time dummy, following changes in IR (Switzerland, SNB)

*** PI dummy is equal to one in the period of positive interest rates

gen PI=(Obs<=378)

*** NI dummy is equal to one in the period of negative interest rates (23 December 
*> 2014, onwards).

gen NI=(Obs>=379)

*** QE2 dummy is equal to one to the recent QE in the U.S.

gen QE2=(Obs>=620)

*** Unit root tests

clemao1 ER
putexcel B2=rscalars using results, sheet("CLEMAO") modify
clemao1 CT
putexcel C2=rscalars using results, sheet("CLEMAO") modify

*** Correlation matrix

pwcorr $logvar, sig

*** Histogram graphs

foreach var of varlist $logvar {
quietly: histogram `var', freq normal saving(`var', replace)
quietly: graph export "l`var'.pdf", replace
}

*** Combine graphs

quietly: gr combine lCT.gph lER.gph IRD.gph lVIX.gph lSM.gph lSMUS.gph, saving(histogram_log_all, replace)
quietly: graph export "histogram_log_all.pdf", replace

*** Closing log

save Granger.dta, replace

saveold Granger_stata13.dta, replace

save "$data\Granger.dta", replace

log close
